14  /**
15   * Proxies classes to provide singleton instances of their methods' return values.
16   */
17  public class Singletoniser {
18  	/**
19  	 * Creates a proxy that implements an interface and subclasses a concrete instance of that interface. The proxy
20  	 * returns the same return value on successive invocations to the same method.
21  	 *
22  	 * <pre>public class SingletoniserExample {
23  	public interface Module {
24  		double random();
25  		double anotherRandom();
26  	}
27  
28  	public static class ModuleImplementation implements Module {
29  		{@literal @}Override public double random() {
30  			return Math.random();
31  		}
32  
33  		{@literal @}Override public double anotherRandom() {
34  			return random();
35  		}
36  	}
37  
38  	public static void main(String[] args) {
39  		Module module = Singletoniser.singletonise(Module.class, new ModuleImplementation());
40  		System.out.println(module.random());
41  		System.out.println(module.random());
42  		System.out.println(module.anotherRandom());
43  		// all invocations print the same value
44  	}
45  }</pre>
46  	 * <p>All interface methods must have no parameters and have a non-void return type.
47  	 * The methods implemented by the implementation must be non-final.
48  	 *
49  	 * <p>A singleton object is cached the first time the method is invoked.
50  	 * The method may be invoked from another object or from the proxied object itself.
51  	 * Access to the singleton cache is synchronised, hence objects returned by this method are thread-safe.
52  	 *
53  	 * @param iface the interface to proxy
54  	 * @param implementation a concrete instance of the interface
55  	 * @return the singletonised implementation
56  	 * @throws SingletoniseException
57  	 *             if the module could not be singletonised
58  	 */
59  	public static <T> T singletonise(final Class<T> iface, final T implementation) {
60  		checkAllMethodsHaveNonVoidReturnType(iface);
61  		checkAllMethodsTakeNoParameters(iface);
62  		checkAllImplementedMethodsAreNonFinal(iface, implementation);
63  
64  		MethodHandler methodHandler = new MethodHandler() {
65  			private final Map<Method, Object> singletons = new HashMap<Method, Object>();
66  
67  			@Override public Object invoke(Object self, Method thisMethod, Method proceed, Object[] args) throws Throwable {
68  				synchronized (singletons) {
69  					if (!singletons.containsKey(thisMethod)) {
70  						singletons.put(thisMethod, proceed.invoke(self, args));
71  					}
72  					return singletons.get(thisMethod);
73  				}
74  			}
75  		};
76  
77  		MethodFilter methodFilter = new MethodFilter() {
78  			@Override public boolean isHandled(Method candidateMethod) {
79  				for (Method interfaceMethod : iface.getMethods()) {
80  					if (areSame(candidateMethod, interfaceMethod)) {
81  						return true;
82  					}
83  				}
84  				return false;
85  			}
86  		};
87  
88  		return createProxy(implementation, methodHandler, methodFilter);
89  	}

91  	private static <T> T createProxy(T implementation, MethodHandler methodHandler, MethodFilter methodFilter) {
92  		try {
93  			ProxyFactory proxyFactory = new ProxyFactory();
94  			proxyFactory.setSuperclass(implementation.getClass());
95  			proxyFactory.setFilter(methodFilter);
96  			Class<?> clazz = proxyFactory.createClass();
97  			@SuppressWarnings("unchecked") T proxy = (T) clazz.newInstance();
98  			((Proxy) proxy).setHandler(methodHandler);
99  			return proxy;
100 		} catch (InstantiationException e) {
101 			throw new SingletoniseException(e);
102 		} catch (IllegalAccessException e) {
103 			throw new SingletoniseException(e);
104 		}
105 	}

107 	private static <T> void checkAllMethodsHaveNonVoidReturnType(Class<T> iface) {
108 		for (Method method : iface.getMethods()) {
109 			if (method.getReturnType() == void.class) {
110 				throw new SingletoniseException(iface.getCanonicalName() + " declares method " + method.getName() + "() with void return type");
111 			}
112 		}
113 	}
114 
115 	private static <T> void checkAllMethodsTakeNoParameters(Class<T> iface) {
116 		for (Method method : iface.getMethods()) {
117 			if (method.getParameterTypes().length != 0) {
118 				throw new SingletoniseException(iface.getCanonicalName() + " declares method " + method.getName() + "() with non-empty argument list");
119 			}
120 		}
121 	}
122 
123 	private static <T> void checkAllImplementedMethodsAreNonFinal(Class<T> iface, T implementation) {
124 		try {
125 			for (Method ifaceMethod : iface.getMethods()) {
126 				if (Modifier.isFinal(implementation.getClass().getMethod(ifaceMethod.getName(), ifaceMethod.getParameterTypes()).getModifiers())) {
127 					throw new SingletoniseException(implementation.getClass().getCanonicalName() + " declares method " + ifaceMethod.getName() + "() as final");
128 				}
129 			}
130 		} catch (NoSuchMethodException e) {
131 			throw new SingletoniseException(e);
132 		}
133 	}

135 	private static boolean areSame(Method m1, Method m2) {
136 		return m1.getName().equals(m2.getName()) && Arrays.equals(m1.getParameterTypes(), m2.getParameterTypes());
137 	}
